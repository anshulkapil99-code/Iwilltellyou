    <script>
        // YOUR UPDATED MESSAGES
        const messages = [
            "Let me be your Man, Let me love you",
            "I fuck you till you go breathless and would love you forever",
            "I would make you moan and scream..",
            "I make you cum and I'll fuck you through it",
            "I'll hold you and kiss you and suck your tits empty",
            "I'll eat you love and make you go crazy.....",
            "You know I love you, You I want you..❤️"
        ];

        let currentIndex = 0;
        let isTransitioning = false;
        
        const canvas = document.getElementById('scratch');
        const ctx = canvas.getContext('2d');
        const status = document.getElementById('status');
        const music = document.getElementById('bg-music');
        const msgLayer = document.getElementById('msg');

        // Draw foil texture on canvas
        function prepareFoil() {
            ctx.globalCompositeOperation = 'source-over';
            // Dark visible foil texture
            const grad = ctx.createLinearGradient(0, 0, canvas.width, canvas.height);
            grad.addColorStop(0, '#0a0a0a');
            grad.addColorStop(0.5, '#222222');
            grad.addColorStop(1, '#0a0a0a');
            ctx.fillStyle = grad;
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            // Texture lines
            ctx.strokeStyle = "rgba(255,255,255,0.06)";
            for(let i=0; i<canvas.width; i+=6) {
                ctx.beginPath(); ctx.moveTo(i, 0); ctx.lineTo(i + 20, canvas.height); ctx.stroke();
            }
        }

        // Access Logic
        function checkAccess() {
            const val = document.getElementById('userNameInput').value.toLowerCase().trim();
            if (val === 'beba') {
                startExperience();
            } else {
                document.getElementById('userNameInput').value = '';
                document.getElementById('userNameInput').placeholder = 'Not for you...';
            }
        }

        // Keyboard Enter Support
        document.getElementById('userNameInput').addEventListener('keypress', (e) => {
            if (e.key === 'Enter') checkAccess();
        });

        function startExperience() {
            // 1. Prepare the scratch card surface FIRST
            prepareFoil();
            
            // 2. Play Audio
            music.play().catch(e => console.log("song.mp3 missing."));

            // 3. Set the text (it's hidden by canvas)
            msgLayer.innerText = messages[currentIndex];

            // 4. Transition Overlay
            const overlay = document.getElementById('overlay');
            overlay.style.opacity = '0';
            
            setTimeout(() => {
                overlay.style.display = 'none';
                const stage = document.getElementById('stage');
                stage.style.display = 'flex';
                setTimeout(() => {
                    stage.style.opacity = '1';
                }, 50);
            }, 800);
        }

        function checkScratch() {
            if (isTransitioning) return;
            const pixels = ctx.getImageData(0, 0, canvas.width, canvas.height).data;
            let cleared = 0;
            // Scan alpha channel
            for (let i = 3; i < pixels.length; i += 4) {
                if (pixels[i] === 0) cleared++;
            }
            const percent = (cleared / (canvas.width * canvas.height)) * 100;
            
            // Trigger at 80% scratched
            if (percent > 80) {
                isTransitioning = true;
                canvas.style.opacity = '0'; // Reveal the rest instantly
                status.innerHTML = "Savor the moment. Next in 5s...";
                
                setTimeout(() => {
                    currentIndex = (currentIndex + 1) % messages.length;
                    resetCard();
                }, 5000); // 5-second delay
            }
        }

        function resetCard() {
            // Reset visibility
            canvas.style.opacity = '1';
            msgLayer.style.opacity = '0';
            
            setTimeout(() => {
                prepareFoil(); // Paint the new black cover
                msgLayer.innerText = messages[currentIndex]; // Change text
                msgLayer.style.opacity = '1';
                isTransitioning = false;
                status.innerText = "Touch the darkness...";
            }, 500);
        }

        let drawing = false;
        const scratch = (e) => {
            if (!drawing || isTransitioning) return;
            const rect = canvas.getBoundingClientRect();
            const x = (e.clientX || (e.touches ? e.touches[0].clientX : 0)) - rect.left;
            const y = (e.clientY || (e.touches ? e.touches[0].clientY : 0)) - rect.top;
            
            ctx.globalCompositeOperation = 'destination-out';
            ctx.beginPath();
            ctx.arc(x, y, 90, 0, Math.PI * 2); // LARGE brush for easy reveal
            ctx.fill();
            checkScratch();
        };

        canvas.addEventListener('mousedown', () => drawing = true);
        canvas.addEventListener('touchstart', (e) => { e.preventDefault(); drawing = true; });
        window.addEventListener('mouseup', () => drawing = false);
        window.addEventListener('touchend', () => drawing = false);
        canvas.addEventListener('mousemove', scratch);
        canvas.addEventListener('touchmove', (e) => { e.preventDefault(); scratch(e); });

        // Hearts particles in loop
        setInterval(() => {
            const container = document.getElementById('heart-container');
            const heart = document.createElement('div');
            heart.className = 'heart';
            heart.innerHTML = '&#10084;';
            heart.style.color = Math.random() > 0.5 ? '#ffffff' : '#000000';
            heart.style.left = Math.random() * 100 + 'vw';
            heart.style.fontSize = (Math.random() * 20 + 20) + 'px';
            container.appendChild(heart);
            setTimeout(() => heart.remove(), 6000);
        }, 300);
    </script>
